/**
 * FIRESTORE SECURITY RULES
 * 
 * Diese Regeln sollten in der Firebase Console unter "Firestore Database" > "Rules" eingegeben werden
 * Copy & Paste den gesamten Inhalt in den Rules Editor
 */

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Authentifizierte Benutzer
    function isSignedIn() {
      return request.auth != null;
    }
    
    // Ist der Benutzer der Besitzer des Dokuments?
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Ist das Projekt öffentlich?
    function isPublic(resource) {
      return resource.data.isPublic == true || resource.data.visibility == 'public';
    }
    
    // ===== BENUTZER RULES =====
    match /users/{userId} {
      allow read, write: if isOwner(userId);
      
      match /profile {
        allow read: if true;
        allow write: if isOwner(userId);
      }
    }
    
    // ===== PROJEKTE RULES =====
    match /projects/{projectId} {
      // Lesen: Nur Owner oder wenn öffentlich
      allow read: if isOwner(resource.data.userId) || isOwner(resource.data.owner) || isPublic(resource);
      
      // Erstellen: Nur authentifizierte Benutzer
      allow create: if isSignedIn();
      
      // Aktualisieren: Nur Owner
      allow update: if isOwner(resource.data.userId) || isOwner(resource.data.owner);
      
      // Löschen: Nur Owner
      allow delete: if isOwner(resource.data.userId) || isOwner(resource.data.owner);
      
      // Finds Subkollektion
      match /finds/{findId} {
        // Allow read if user can access the project (owner or public)
        allow read: if isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.userId) || isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.owner) || isPublic(get(/databases/$(database)/documents/projects/$(projectId)));
        allow create: if isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.userId) || isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.owner);
        allow update, delete: if isOwner(resource.data.creator) || isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.userId) || isOwner(get(/databases/$(database)/documents/projects/$(projectId)).data.owner);
      }
    }
    
    // ===== FUNDE RULES (Toplevel) =====
    match /finds/{findId} {
      // Allow read if user is owner (creator) or the associated project is public
      allow read: if isOwner(resource.data.creator) || (
        'projectId' in resource.data && 
        (isOwner(get(/databases/$(database)/documents/projects/$(resource.data.projectId)).data.userId) || 
         isOwner(get(/databases/$(database)/documents/projects/$(resource.data.projectId)).data.owner) ||
         isPublic(get(/databases/$(database)/documents/projects/$(resource.data.projectId))))
      );
      
      // Erstellen: Nur authentifizierte Benutzer
      allow create: if isSignedIn();
      
      // Aktualisieren und Löschen: Nur Owner (creator) oder Project Owner
      allow update, delete: if isOwner(resource.data.creator) || (
        'projectId' in resource.data && 
        (isOwner(get(/databases/$(database)/documents/projects/$(resource.data.projectId)).data.userId) ||
         isOwner(get(/databases/$(database)/documents/projects/$(resource.data.projectId)).data.owner))
      );
    }
    
    // ===== AUSGRABUNGSSTÄTTEN RULES =====
    match /excavationSites/{siteId} {
      allow read: if true;
      allow create, update: if isSignedIn();
      allow delete: if false;
    }
    
    // ===== DEFAULT (alles andere verbieten) =====
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
